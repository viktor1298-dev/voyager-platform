```typescript
function computePercentDomain(data: MetricsDataPoint[], metrics: MetricKey[]): [number, number] {
  let min = Infinity
  let max = -Infinity
  for (const point of data) {
    for (const key of metrics) {
      const cfg = METRIC_CONFIG[key]
      if (cfg.yAxis !== 'percent') continue
      const val = point[cfg.dataKey]
      if (typeof val === 'number' && !Number.isNaN(val)) {
        if (val < min) min = val
        if (val > max) max = val
      }
    }
  }
  if (!Number.isFinite(min) || !Number.isFinite(max)) return [0, 100]
  // Add 10% padding, floor min to 0, ceil max to nearest 10 (min range 10)
  const paddedMin = Math.max(0, Math.floor((min - (max - min) * 0.1) / 5) * 5)
  const paddedMax = Math.min(100, Math.ceil((max + (max - min) * 0.1) / 5) * 5)
  if (paddedMax - paddedMin < 10) return [Math.max(0, paddedMin - 5), Math.min(100, paddedMax + 5)]
  return [paddedMin, paddedMax]
}
```

Use it in the percent YAxis:
```tsx
const [yMin, yMax] = computePercentDomain(chartData, activeMetrics)
// ...
<YAxis yAxisId="percent" domain={[yMin, yMax]} tickCount={5} ... />
```
Remove the fixed `ticks={[0, 25, 50, 75, 100]}` prop — let Recharts auto-generate ticks from `tickCount={5}`.

**4. X-axis adaptive formatting (MetricsAreaChart.tsx) — per STYLE-04:**
Replace the `formatXAxis` function with range-aware formatting:
```typescript
function formatXAxis(iso: string, range: MetricsRange): string {
  const d = new Date(iso)
  if (!iso || Number.isNaN(d.getTime())) return ''

  switch (range) {
    case '5m':
    case '15m':
      // Short ranges: HH:MM:SS for second-level precision
      return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit' })
    case '30m':
    case '1h':
    case '3h':
    case '6h':
    case '12h':
    case '24h':
      // Medium ranges: HH:MM (no seconds)
      return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
    case '2d':
    case '7d':
      // Multi-day: Mon Day (abbreviated weekday + day number)
      return d.toLocaleDateString([], { month: 'short', day: 'numeric' })
    default:
      return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
  }
}
```
