```typescript
export default function ServicesPage() {
  usePageTitle('Cluster Services')
  const { id: routeSegment } = useParams<{ id: string }>()
  const clusterId = getClusterIdFromRouteSegment(routeSegment)
  const dbCluster = trpc.clusters.get.useQuery({ id: clusterId })
  const resolvedId = dbCluster.data?.id ?? clusterId
  const hasCredentials = Boolean((dbCluster.data as Record<string, unknown> | undefined)?.hasCredentials)

  const servicesQuery = trpc.services.listDetail.useQuery(
    { clusterId: resolvedId },
    { enabled: hasCredentials && !!resolvedId, refetchInterval: 30000 },
  )

  return (
    <ResourcePageScaffold<ServiceDetail>
      title="Cluster Services"
      data={(servicesQuery.data ?? []) as ServiceDetail[]}
      isLoading={dbCluster.isLoading || servicesQuery.isLoading}
      hasCredentials={hasCredentials}
      getNamespace={(svc) => svc.namespace}
      getKey={(svc) => `${svc.namespace}/${svc.name}`}
      filterFn={(svc, q) =>
        svc.name.toLowerCase().includes(q) ||
        svc.namespace.toLowerCase().includes(q) ||
        svc.type.toLowerCase().includes(q)
      }
      renderSummary={(svc) => <ServiceSummary svc={svc} />}
      renderDetail={(svc) => <ServiceExpandedDetail svc={svc} />}
      searchPlaceholder="Filter services by name, namespace, or type..."
      emptyMessage="No services found in this cluster"
      emptyIcon={<Globe className="h-8 w-8 text-[var(--color-text-dim)]" />}
    />
  )
}
```
