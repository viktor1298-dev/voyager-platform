Build the full K8s Watch real-time system: backend ResourceWatchManager with informers for 15 resource types, SSE endpoint for streaming resource changes, and a client-side hook placed in the cluster layout that auto-invalidates tRPC query cache on resource changes. Purpose: Enable real-time resource updates across ALL cluster tabs without any per-tab code. When a pod is created, deployment scaled, or configmap updated in the cluster, the UI reflects it within ~1s.

**Step 4: Create apps/api/src/lib/resource-watch-manager.ts:**

New file. Reference-counted manager for 12 additional resource type informers (pods/deployments/nodes are already watched by ClusterWatchManager — but we re-emit their events as ResourceChangeEvent for the unified stream). The ResourceWatchManager watches: services, ingresses, statefulsets, daemonsets, jobs, cronjobs, hpa, configmaps, secrets, pvcs, namespaces, events.

```typescript
import * as k8s from '@kubernetes/client-node'
import type { ResourceChangeEvent, ResourceType } from '@voyager/types'
import { clusterClientPool } from './cluster-client-pool.js'
import { voyagerEmitter } from './event-emitter.js'

// Reference count per cluster
const subscriberCounts = new Map<string, Set<string>>()

interface ResourceInformerSet {
  informers: k8s.Informer<k8s.KubernetesObject>[]
}

class ResourceWatchManager {
  private clusters = new Map<string, ResourceInformerSet>()

  subscribe(clusterId: string, connectionId: string): void {
    let subs = subscriberCounts.get(clusterId)
    if (!subs) {
      subs = new Set()
      subscriberCounts.set(clusterId, subs)
    }
    subs.add(connectionId)
    if (subs.size === 1) {
      this.startWatches(clusterId).catch(err =>
        console.error(`[ResourceWatchManager] Failed to start watches for ${clusterId}:`, err)
      )
    }
  }

  unsubscribe(clusterId: string, connectionId: string): void {
    const subs = subscriberCounts.get(clusterId)
    if (!subs) return
    subs.delete(connectionId)
    if (subs.size === 0) {
      subscriberCounts.delete(clusterId)
      this.stopWatches(clusterId)
    }
  }

  private async startWatches(clusterId: string): Promise<void> { ... }
  private stopWatches(clusterId: string): void { ... }
  stopAll(): void { ... }
}
```

In `startWatches`, create informers for each resource type:
- CoreV1Api: services (`/api/v1/services`, `listServiceForAllNamespaces`), configmaps (`/api/v1/configmaps`, `listConfigMapForAllNamespaces`), secrets (`/api/v1/secrets`, `listSecretForAllNamespaces`), pvcs (`/api/v1/persistentvolumeclaims`, `listPersistentVolumeClaimForAllNamespaces`), namespaces (`/api/v1/namespaces`, `listNamespace`), events (`/api/v1/events`, `listEventForAllNamespaces`)
- AppsV1Api: statefulsets (`/apis/apps/v1/statefulsets`, `listStatefulSetForAllNamespaces`), daemonsets (`/apis/apps/v1/daemonsets`, `listDaemonSetForAllNamespaces`)
- BatchV1Api: jobs (`/apis/batch/v1/jobs`, `listJobForAllNamespaces`), cronjobs (`/apis/batch/v1/cronjobs`, `listCronJobForAllNamespaces`)

**Step 1: Create apps/web/src/hooks/useResourceSSE.ts:**

```typescript
'use client'
import { useEffect, useRef } from 'react'
import { trpc } from '@/lib/trpc'
import type { ResourceChangeEvent, ResourceType } from '@voyager/types'

// Map ResourceType to tRPC router invalidation
const RESOURCE_INVALIDATION_MAP: Record<ResourceType, string> = {
  pods: 'pods',
  deployments: 'deployments',
  statefulsets: 'statefulSets',
  daemonsets: 'daemonSets',
  services: 'services',
  ingresses: 'ingresses',
  jobs: 'jobs',
  cronjobs: 'cronJobs',
  hpa: 'hpa',
  configmaps: 'configMaps',
  secrets: 'secrets',
  pvcs: 'pvcs',
  namespaces: 'namespaces',
  events: 'events',
  nodes: 'nodes',
}

export function useResourceSSE(clusterId: string | null) {
  const utils = trpc.useUtils()
  const eventSourceRef = useRef<EventSource | null>(null)
  const debounceTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null)
  const pendingInvalidations = useRef<Set<string>>(new Set())

  useEffect(() => {
    if (!clusterId) return

    const apiUrl = process.env.NEXT_PUBLIC_API_URL || ''
    const url = `${apiUrl}/api/resources/stream?clusterId=${clusterId}`
    const es = new EventSource(url, { withCredentials: true })
    eventSourceRef.current = es

    es.addEventListener('resource-change', (e: MessageEvent) => {
      try {
        const events: ResourceChangeEvent[] = JSON.parse(e.data)
        for (const evt of events) {
          const routerKey = RESOURCE_INVALIDATION_MAP[evt.resourceType]
          if (routerKey) pendingInvalidations.current.add(routerKey)
        }
        // Debounce: batch invalidations within 1s window
        if (!debounceTimerRef.current) {
          debounceTimerRef.current = setTimeout(() => {
            for (const key of pendingInvalidations.current) {
              // Invalidate the list query for the changed resource type
              ;(utils as Record<string, any>)[key]?.list?.invalidate?.({ clusterId })
            }
            pendingInvalidations.current.clear()
            debounceTimerRef.current = null
          }, 1000)
        }
      } catch { /* ignore parse errors */ }
    })

    es.onerror = () => {
      // EventSource auto-reconnects — just log
      console.warn('[useResourceSSE] Connection error, auto-reconnecting...')
    }

    return () => {
      es.close()
      eventSourceRef.current = null
      if (debounceTimerRef.current) {
        clearTimeout(debounceTimerRef.current)
        debounceTimerRef.current = null
      }
      pendingInvalidations.current.clear()
    }
  }, [clusterId, utils])
}
```

Key details:
- Uses EventSource (SSE) with `withCredentials: true` for cookie auth
- Debounces invalidations with 1s window to prevent UI thrashing during rolling deployments
- Uses `trpc.useUtils()` for cache invalidation — same pattern as existing pod delete mutation
- EventSource auto-reconnects on error (built-in browser behavior)
- Does NOT use `typeof window` check in render (Gotcha #13) — the hook is only called in client component
