**Deployments page rewrite** (`deployments/page.tsx`):

1. Keep the existing `DeploymentDetail` interface and `statusColor` helper function.
2. Replace the entire page render with ResourcePageScaffold:
```typescript
import { ResourcePageScaffold } from '@/components/resource'
import { ConditionsList, DetailTabs, TagPills } from '@/components/expandable'
import { Rocket } from 'lucide-react'

export default function DeploymentsPage() {
  usePageTitle('Deployments')
  const { id: routeSegment } = useParams<{ id: string }>()
  const clusterId = getClusterIdFromRouteSegment(routeSegment)
  const dbCluster = trpc.clusters.get.useQuery({ id: clusterId })
  const resolvedId = dbCluster.data?.id ?? clusterId
  const hasCredentials = Boolean((dbCluster.data as Record<string, unknown>)?.hasCredentials)
  const query = trpc.deployments.list.useQuery(
    { clusterId: resolvedId },
    { enabled: hasCredentials && !!resolvedId, staleTime: 30_000 }
  )

  return (
    <ResourcePageScaffold<DeploymentDetail>
      title="Deployments"
      icon={<Rocket className="h-5 w-5" />}
      queryResult={query}
      getNamespace={(d) => d.namespace}
      getKey={(d) => `${d.namespace}/${d.name}`}
      filterFn={(d, q) =>
        d.name.toLowerCase().includes(q) ||
        d.namespace.toLowerCase().includes(q) ||
        d.status.toLowerCase().includes(q)
      }
      renderSummary={(d) => <DeploymentSummary d={d} />}
      renderDetail={(d) => <DeploymentExpandedDetail d={d} />}
      searchPlaceholder="Search deployments..."
      emptyMessage="No deployments found"
    />
  )
}
```

3. Create `DeploymentSummary` component showing horizontally: name (bold), ready count badge (`{readyReplicas}/{replicas}`), status pill with color, image name (truncated), age.
4. Keep existing `DeploymentExpandedDetail` but convert from ExpandableTableRow detail to standalone component that gets passed to `renderDetail`. Keep Replicas, Strategy, Conditions, Labels tabs via `DetailTabs`.
