```typescript
'use client'

import { ExternalLink } from 'lucide-react'
import { useParams, useRouter } from 'next/navigation'

interface CrossResourceLinkProps {
  /** Target tab path segment (e.g., 'pods', 'services', 'jobs') */
  tab: string
  /** Display text for the link */
  label: string
  /** Optional: highlight query param to identify target resource */
  highlight?: Record<string, string>
  /** Optional: icon */
  icon?: ReactNode
}

export function CrossResourceLink({ tab, label, highlight, icon }: CrossResourceLinkProps) {
  const router = useRouter()
  const { id: routeSegment } = useParams<{ id: string }>()

  const handleClick = () => {
    const base = `/clusters/${routeSegment}/${tab}`
    if (highlight) {
      const params = new URLSearchParams({ highlight: JSON.stringify(highlight) })
      router.push(`${base}?${params.toString()}`)
    } else {
      router.push(base)
    }
  }

  return (
    <button
      type="button"
      onClick={handleClick}
      className="inline-flex items-center gap-1 text-[11px] font-mono text-[var(--color-accent)] hover:underline cursor-pointer transition-colors"
    >
      {icon}
      {label}
      <ExternalLink className="h-3 w-3 opacity-50" />
    </button>
  )
}
```

**2. Create RelatedPodsList.tsx — Pods list filtered by label selector:**

Create `apps/web/src/components/resource/RelatedPodsList.tsx`:

```typescript
'use client'

import { Box } from 'lucide-react'
import { useMemo } from 'react'
import { Skeleton } from '@/components/ui/skeleton'
import { trpc } from '@/lib/trpc'
import { CrossResourceLink } from './CrossResourceLink'

interface RelatedPodsListProps {
  clusterId: string
  /** Label selectors to match pods against (e.g., { app: 'nginx', version: 'v1' }) */
  matchLabels: Record<string, string>
  /** Whether this tab is currently active (controls query enabled) */
  enabled: boolean
}

export function RelatedPodsList({ clusterId, matchLabels, enabled }: RelatedPodsListProps) {
  // Reuse the existing pods.list query (it returns all pods with labels)
  const podsQuery = trpc.pods.list.useQuery(
    { clusterId },
    { enabled, staleTime: 30_000 },
  )

  const matchingPods = useMemo(() => {
    if (!podsQuery.data) return []
    const pods = podsQuery.data as { name: string; namespace: string; status: string; labels: Record<string, string> }[]
    return pods.filter((pod) =>
      Object.entries(matchLabels).every(([k, v]) => pod.labels?.[k] === v)
    )
  }, [podsQuery.data, matchLabels])

  if (!enabled) return null

  if (podsQuery.isLoading) {
    return <div className="space-y-2 p-3">{Array.from({ length: 3 }).map((_, i) => <Skeleton key={i} className="h-6 w-full" />)}</div>
  }

  if (matchingPods.length === 0) {
    return (
      <div className="p-4 text-center">
        <Box className="h-6 w-6 mx-auto text-[var(--color-text-dim)] mb-2" />
        <p className="text-[11px] text-[var(--color-text-muted)]">No matching pods found</p>
      </div>
    )
  }

  return (
    <div className="space-y-1.5 p-3">
      <p className="text-[10px] font-semibold uppercase tracking-wide text-[var(--color-text-muted)] mb-2">
        {matchingPods.length} pod{matchingPods.length !== 1 ? 's' : ''} matching selector
      </p>
      {matchingPods.map((pod) => (
        <div key={`${pod.namespace}/${pod.name}`} className="flex items-center justify-between py-1.5 px-2 rounded-md bg-white/[0.02] hover:bg-white/[0.04] transition-colors">
          <div className="flex items-center gap-2 min-w-0">
            <StatusDot status={pod.status} />
            <span className="text-[11px] font-mono text-[var(--color-text-primary)] truncate">{pod.name}</span>
            <span className="text-[10px] text-[var(--color-text-dim)]">{pod.namespace}</span>
          </div>
          <CrossResourceLink tab="pods" label="View" highlight={{ name: pod.name, namespace: pod.namespace }} />
        </div>
      ))}
    </div>
  )
}

function StatusDot({ status }: { status: string }) {
  const color = status === 'Running' ? 'bg-emerald-400' : status === 'Pending' ? 'bg-amber-400' : status === 'Failed' ? 'bg-red-400' : 'bg-gray-400'
  return <span className={`inline-block h-1.5 w-1.5 rounded-full ${color}`} />
}
```

In `apps/web/src/app/clusters/[id]/pods/page.tsx`, the `DeletePodDialog` already calls `utils.pods.list.invalidate({ clusterId })` on success (line 537). Verify this is present. If it is, add **optimistic removal** — before the mutation resolves, immediately filter the deleted pod from the cache:

In `DeletePodDialog`, update the mutation:
```typescript
const deleteMutation = trpc.pods.delete.useMutation({
  onMutate: async ({ podName, namespace }) => {
    // Cancel outgoing refetches
    await utils.pods.list.cancel({ clusterId })
    // Snapshot previous data
    const previous = utils.pods.list.getData({ clusterId })
    // Optimistically remove the pod
    utils.pods.list.setData({ clusterId }, (old) =>
      old ? (old as PodData[]).filter(p => !(p.name === podName && p.namespace === namespace)) : old
    )
    return { previous }
  },
  onSuccess: () => {
    toast.success(`Pod ${pod.name} deleted`)
    utils.pods.list.invalidate({ clusterId })
    onClose()
  },
  onError: (err, _vars, context) => {
    // Rollback on error
    if (context?.previous) {
      utils.pods.list.setData({ clusterId }, context.previous)
    }
    toast.error(`Failed to delete pod: ${err.message}`)
  },
})
```

This provides immediate visual feedback (pod disappears from list) while the actual deletion happens in the background.
