**CrossResourceNav.tsx** — Utility functions for cross-resource navigation:

```typescript
'use client'
import { useRouter, useParams } from 'next/navigation'
import { getClusterRouteSegment } from '@/components/cluster-route'

// Navigate to a specific resource on a specific tab
export function useResourceNavigation() {
  const router = useRouter()
  const { id: routeSegment } = useParams<{ id: string }>()

  return {
    navigateToResource: (tab: string, resourceKey: string) => {
      // tab: 'pods', 'services', 'deployments', etc.
      // resourceKey: 'namespace/name' to highlight
      const highlight = encodeURIComponent(resourceKey)
      router.push(`/clusters/${routeSegment}/${tab}?highlight=${highlight}`)
    },
    navigateToTab: (tab: string) => {
      router.push(`/clusters/${routeSegment}/${tab}`)
    },
  }
}
```

```typescript
const deleteMutation = trpc.pods.delete.useMutation({
  onMutate: async ({ name, namespace }) => {
    await utils.pods.list.cancel({ clusterId })
    const previous = utils.pods.list.getData({ clusterId })
    utils.pods.list.setData({ clusterId }, (old) =>
      old?.filter(p => !(p.name === name && p.namespace === namespace))
    )
    return { previous }
  },
  onError: (err, _, context) => {
    if (context?.previous) utils.pods.list.setData({ clusterId }, context.previous)
    toast.error(`Failed to delete pod: ${err.message}`)
  },
  onSuccess: () => {
    toast.success(`Pod deleted`)
    utils.pods.list.invalidate({ clusterId })
  },
})
```
